function jsonOut(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

function ss() { return SpreadsheetApp.getActiveSpreadsheet(); }
function sh(name) { return ss().getSheetByName(name) || ss().insertSheet(name); }

function ensureSheet(name, headers) {
  const sheet = sh(name);
  const range = sheet.getDataRange();
  const values = range.getValues();
  if(values.length === 0 || values[0].every(v => String(v || "").trim() === "")) {
    sheet.clear();
    sheet.appendRow(headers);
    return sheet;
  }

  const current = values[0].map(h => String(h).trim());
  headers.forEach(h => {
    if(current.indexOf(h) < 0){
      sheet.getRange(1, current.length + 1).setValue(h);
      current.push(h);
    }
  });
  return sheet;
}


function rowsAsObjects(sheetName) {
  const sheet = sh(sheetName);
  const values = sheet.getDataRange().getValues();
  if (values.length === 0) return { sheet, headers: [], rows: [] };
  const headers = values[0].map(h => String(h).trim());
  const rows = values.slice(1).filter(r => r.some(c => c !== "")).map((row, i) => {
    const o = { __rowNumber: i + 2 };
    headers.forEach((h, idx) => o[h] = row[idx]);
    return o;
  });
  return { sheet, headers, rows };
}

function col(headers, name) { return headers.findIndex(h => String(h).trim() === name) + 1; }
function norm(v) { return String(v || "").trim(); }
function lower(v) { return norm(v).toLowerCase(); }
function isTrue(v) { return lower(v) === "true" || v === true; }
function isDone(v) { return ["concluído", "concluido", "finalizado"].includes(lower(v)); }
function now() { return new Date(); }

function nextId(rows, field) {
  return rows.reduce((m, r) => Math.max(m, Number(r[field] || 0)), 0) + 1;
}

function setCell(sheet, headers, rowNumber, field, value) {
  const c = col(headers, field);
  if (c > 0) sheet.getRange(rowNumber, c).setValue(value);
}

function ensureColumn(sheetName, field) {
  const obj = rowsAsObjects(sheetName);
  if (obj.headers.indexOf(field) >= 0) return rowsAsObjects(sheetName);
  const c = obj.headers.length + 1;
  obj.sheet.getRange(1, c).setValue(field);
  return rowsAsObjects(sheetName);
}

function setCellSafe(sheetName, rowNumber, field, value) {
  const obj = ensureColumn(sheetName, field);
  setCell(obj.sheet, obj.headers, rowNumber, field, value);
}

function gerarMatricula(rows) {
  const existentes = rows.map(r => norm(r.matricula));
  let mat = "";
  do {
    mat = String.fromCharCode(65 + Math.floor(Math.random() * 26)) + Math.floor(1000 + Math.random() * 9000);
  } while (existentes.includes(mat));
  return mat;
}

function login(matricula) {
  const { sheet, headers, rows } = rowsAsObjects(SHEETS.operadores);
  const user = rows.find(r => norm(r.matricula) === norm(matricula));
  if (!user) return jsonOut(null);
  if (!isTrue(user.ativo)) return jsonOut({erro:"Usuário inativo"});
  setCell(sheet, headers, user.__rowNumber, "ultimo_login_em", now());
  return jsonOut(user);
}

}

function codigoOS(id) { return "OS-" + new Date().getFullYear() + "-" + String(id).padStart(4, "0"); }

function criarOS(d) {
  const osData = rowsAsObjects(SHEETS.os);
  const id = nextId(osData.rows, "id_os");
  const cod = codigoOS(id);
  const modeloId = d.id_modelo || "";
  osData.sheet.appendRow([id, cod, d.motor, now(), "Em processo", d.setor_inicial, d.setor_inicial, d.criado_por || d.operador_nome || "", d.criado_por_matricula || "", "", false, false, modeloId, "", d.observacao || "", "", "", now()]);
  criarChecklistPrincipalPorModelo(id, d.setor_inicial, modeloId);
  addHistorico("OS_CRIADA", d.criado_por || d.operador_nome, d.criado_por_matricula, cod, d.setor_inicial, d.motor);
  addLog("OS_CRIADA", d.criado_por || d.operador_nome, d.motor, cod);
  return {sucesso:true, id_os:id, codigo_os:cod};
}

function criarChecklistPrincipalPorModelo(idOS, setor, idModelo) {
  const check = rowsAsObjects(SHEETS.checklist);
  const modelos = rowsAsObjects(SHEETS.modelos).rows
    .filter(m => isTrue(m.ativo) && lower(m.tipo) === "principal" && norm(m.setor) === norm(setor))
    .sort((a,b)=>Number(a.ordem||0)-Number(b.ordem||0));
  const lista = modelos.length ? modelos : [{descricao:"Executar checklist do setor", ordem:1}];
  let idItem = nextId(check.rows, "id_item");
  lista.forEach((m, idx) => {
    check.sheet.appendRow([idItem++, idOS, "Principal", "", setor, setor, m.descricao, false, idx + 1, "", "", ""]);
  });
}

function calcularProgresso(idOS) {
  const subtRows = rowsAsObjects(SHEETS.subtarefas).rows;
  const subtExcluidas = subtRows
    .filter(s => String(s.id_os) === String(idOS) && isTrue(s.excluida))
    .map(s => String(s.id_subtarefa));

  const checklist = rowsAsObjects(SHEETS.checklist).rows
    .filter(i => String(i.id_os) === String(idOS))
    .filter(i => !(i.id_subtarefa && subtExcluidas.indexOf(String(i.id_subtarefa)) >= 0));

  const total = checklist.length;
  const concluidos = checklist.filter(i => isTrue(i.concluido)).length;
  const principal = checklist.filter(i => norm(i.origem) === "Principal");
  const sub = checklist.filter(i => norm(i.origem) === "Subtarefa");

  const subtarefasAtivas = subtRows.filter(s => String(s.id_os) === String(idOS) && !isTrue(s.excluida));
  const subtPendentes = subtarefasAtivas.filter(s => !isDone(s.status)).length;

  return {
    total_itens: total,
    total_concluidos: concluidos,
    checklist_total: principal.length,
    checklist_concluidos: principal.filter(i=>isTrue(i.concluido)).length,
    subtarefas_total: subtarefasAtivas.length,
    subtarefas_concluidas: subtarefasAtivas.filter(s=>isDone(s.status)).length,
    subtarefas_pendentes: subtPendentes,
    percentual_total: total ? Math.round((concluidos/total)*100) : 0,
    atualizado_em: now()
  };
}

function listarOS(setor, matricula) {
  const setorNorm = norm(setor);
  const osRows = rowsAsObjects(SHEETS.os).rows
    .filter(o => !isTrue(o.excluida) && !isDone(o.status));

  const subt = rowsAsObjects(SHEETS.subtarefas).rows
    .filter(s => !isTrue(s.excluida) && !isDone(s.status));

  const result = [];
  const seen = {};

  osRows.forEach(o => {
    const prog = calcularProgresso(o.id_os);

    // Card principal: somente setor atual da OS.
    if (norm(o.setor_atual) === setorNorm) {
      const key = "OS:" + o.id_os;
      if(!seen[key]){
        seen[key] = true;
        result.push(Object.assign({}, o, prog, {modo_subtarefa:false, tipo_card:"Principal"}));
      }
    }

    // Card de subtarefa: somente setor destino da subtarefa.
    subt
      .filter(s => String(s.id_os) === String(o.id_os) && norm(s.setor_destino) === setorNorm)
      .forEach(s => {
        const key = "SUB:" + s.id_subtarefa;
        if(seen[key]) return;
        seen[key] = true;

        result.push(Object.assign({}, o, prog, {
          modo_subtarefa:true,
          tipo_card:"Subtarefa",
          id_subtarefa:s.id_subtarefa,
          descricao_subtarefa:s.descricao,
          setor_origem:s.setor_origem,
          setor_destino:s.setor_destino,
          setor_atual:s.setor_destino
        }));
      });
  });

  return jsonOut(result);
}

function buscarOS(termo, setor) {
  const clean = norm(termo);
  const setorNorm = norm(setor);
  const osRows = rowsAsObjects(SHEETS.os).rows.filter(o => !isTrue(o.excluida));
  const os = osRows.find(o => norm(o.codigo_os) === clean || norm(o.codigo_os).endsWith(clean.padStart(4, "0")) || String(o.id_os) === clean);
  if (!os) return jsonOut({erro:"OS não encontrada"});

  const subt = rowsAsObjects(SHEETS.subtarefas).rows
    .filter(s => !isTrue(s.excluida) && !isDone(s.status) && String(s.id_os) === String(os.id_os) && norm(s.setor_destino) === setorNorm);

  if (norm(os.setor_atual) === setorNorm) {
    return jsonOut(Object.assign({}, os, calcularProgresso(os.id_os), {permitido:true, modo_subtarefa:false, tipo_card:"Principal"}));
  }

  if (subt.length > 0) {
    const s = subt[0];
    return jsonOut(Object.assign({}, os, calcularProgresso(os.id_os), {
      permitido:true,
      modo_subtarefa:true,
      tipo_card:"Subtarefa",
      id_subtarefa:s.id_subtarefa,
      descricao_subtarefa:s.descricao,
      setor_origem:s.setor_origem,
      setor_destino:s.setor_destino,
      setor_atual:s.setor_destino
    }));
  }

  return jsonOut(Object.assign({}, os, calcularProgresso(os.id_os), {permitido:false}));
}

function concluirItem(d) {
  const data = rowsAsObjects(SHEETS.checklist);
  const item = data.rows.find(i => String(i.id_item) === String(d.id_item));
  if (!item) return jsonOut({sucesso:false, erro:"Item não encontrado"});

  const setorNorm = norm(d.setor);
  const origem = norm(item.origem);

  if (origem === "Subtarefa") {
    const sub = rowsAsObjects(SHEETS.subtarefas).rows.find(s => String(s.id_subtarefa) === String(item.id_subtarefa));
    if (!sub || isTrue(sub.excluida)) return jsonOut({sucesso:false, erro:"Subtarefa não encontrada ou excluída"});
    if (norm(sub.setor_destino) !== setorNorm) {
      return jsonOut({sucesso:false, erro:"Este item pertence a outro setor"});
    }
  } else {
    const os = rowsAsObjects(SHEETS.os).rows.find(o => String(o.id_os) === String(item.id_os));
    if (!os) return jsonOut({sucesso:false, erro:"OS não encontrada"});
    if (norm(os.setor_atual) !== setorNorm || norm(item.setor || item.etapa) !== setorNorm) {
      return jsonOut({sucesso:false, erro:"Este checklist pertence a outro setor"});
    }
  }

  setCell(data.sheet, data.headers, item.__rowNumber, "concluido", true);
  setCell(data.sheet, data.headers, item.__rowNumber, "concluido_por", d.operador_nome);
  setCell(data.sheet, data.headers, item.__rowNumber, "concluido_em", now());

  ensureColumn(SHEETS.os, "operador_atual_nome");
  ensureColumn(SHEETS.os, "operador_atual_matricula");
  ensureColumn(SHEETS.os, "ultimo_movimento_em");

  const os = rowsAsObjects(SHEETS.os).rows.find(o => String(o.id_os) === String(item.id_os));
  const codigo = os ? os.codigo_os : "";

  if (os) {
    setCellSafe(SHEETS.os, os.__rowNumber, "operador_atual_nome", d.operador_nome || "");
    setCellSafe(SHEETS.os, os.__rowNumber, "operador_atual_matricula", d.matricula || "");
    setCellSafe(SHEETS.os, os.__rowNumber, "ultimo_movimento_em", now());
  }

  addHistorico("ITEM_CONCLUIDO", d.operador_nome, d.matricula, codigo, item.setor, item.descricao);
  addLog("ITEM_CONCLUIDO", d.operador_nome, item.descricao, codigo);

  return jsonOut(Object.assign({sucesso:true}, calcularProgresso(item.id_os)));
}

function isHoje(data) {
  if (!data) return false;
  const d = new Date(data);
  if (isNaN(d.getTime())) return false;
  const h = new Date();
  return d.getFullYear() === h.getFullYear() && d.getMonth() === h.getMonth() && d.getDate() === h.getDate();
}

function listarNotificacoesOperador(matricula, setor) {
  const mat = norm(matricula);
  const setorNorm = norm(setor);
  const rows = rowsAsObjects(SHEETS.historico).rows
    .filter(r => isHoje(r.data_hora))
    .sort((a,b)=>new Date(b.data_hora)-new Date(a.data_hora));

  const relevantes = rows.filter(r => {
    const acao = norm(r.acao).toUpperCase();
    const rMat = norm(r.matricula);
    const rSetor = norm(r.setor);

    // Ações próprias do operador.
    if (rMat && rMat === mat) return true;

    // Demandas novas encaminhadas/listadas para o setor do operador.
    if ((acao === "OS_CRIADA" || acao === "SUBTAREFA_CRIADA") && rSetor === setorNorm) return true;

    // Respostas da administração sobre troca de setor do próprio operador.
    if ((acao === "TROCA_SETOR_APROVADA" || acao === "TROCA_SETOR_NEGADA") && norm(r.detalhes).indexOf(mat) >= 0) return true;

    return false;
  }).slice(0,60);

  return jsonOut(relevantes);
}

function resolverQR(valor, setor, modo, contexto) {
  const v = norm(valor);
  const modoQR = norm(modo || "auto");

  if (modoQR !== "auto" && modoQR !== "kit" && v.startsWith("KIT:")) return jsonOut({erro:"QR de kit bloqueado pelo modo atual"});

  if (v.startsWith("KIT:")) {
    const codigo = v.replace("KIT:", "");
    const kit = garantirKitsQR().rows.find(k => norm(k.codigo_qr) === norm(codigo));
    if (!kit) return jsonOut({erro:"Kit não encontrado"});
    const os = kit.id_os_atual ? rowsAsObjects(SHEETS.os).rows.find(o => String(o.id_os) === String(kit.id_os_atual)) : null;
    if (os) {
      return jsonOut({
        tipo:"KIT",
        mensagem:`Kit ${kit.codigo_qr} vinculado à ${os.codigo_os}`,
        kit:kit,
        os:Object.assign({}, os, calcularProgresso(os.id_os), {kit_qr:kit.codigo_qr})
      });
    }
    return jsonOut({tipo:"KIT", mensagem:`Kit ${kit.codigo_qr} livre`, kit:kit});
  }

  if (modoQR !== "auto" && modoQR !== "subtarefa" && v.startsWith("SUB:")) return jsonOut({erro:"QR de subtarefa bloqueado pelo modo atual"});

  if (v.startsWith("SUB:")) {
    const id = v.split(":")[1];
    const sub = rowsAsObjects(SHEETS.subtarefas).rows.find(s => String(s.id_subtarefa) === String(id));
    if (!sub) return jsonOut({erro:"Subtarefa não encontrada"});
    const os = rowsAsObjects(SHEETS.os).rows.find(o => String(o.id_os) === String(sub.id_os));
    return jsonOut({tipo:"Subtarefa", os:Object.assign({}, os, calcularProgresso(os.id_os), {modo_subtarefa:true, id_subtarefa:sub.id_subtarefa, descricao_subtarefa:sub.descricao})});
  }

  const busca = buscarOSRaw(v);
  if (busca) return jsonOut({tipo:"OS", os:Object.assign({}, busca, calcularProgresso(busca.id_os))});
  return jsonOut({erro:"QR não reconhecido"});
}

}

function etapaInicialDaOS(os, checkRows) {
  const modelos = rowsAsObjects(SHEETS.modelos).rows;
  const modelo = modelos.find(m => String(m.id_modelo) === String(os.id_modelo) && String(m.tipo).toLowerCase() === "principal");
  if (modelo && modelo.setor) return modelo.setor;
  const principais = (checkRows || rowsAsObjects(SHEETS.checklist).rows)
    .filter(i => String(i.id_os) === String(os.id_os) && String(i.origem || "").toLowerCase() === "principal")
    .sort((a,b)=>Number(a.ordem||0)-Number(b.ordem||0));
  if (principais[0] && principais[0].etapa) return principais[0].etapa;
  return "Desmontagem";
}

function republicarOS(d) {
  const osObj = rowsAsObjects(SHEETS.os);
  const original = osObj.rows.find(o => String(o.id_os) === String(d.id_os));
  if (!original) return jsonOut({sucesso:false, erro:"OS original não encontrada"});

  const check = rowsAsObjects(SHEETS.checklist);
  const subs = rowsAsObjects(SHEETS.subtarefas);
  const etapaInicial = etapaInicialDaOS(original, check.rows);

  const novoId = nextId(osObj.rows, "id_os");
  const novoCodigo = codigoOS(novoId);
  osObj.sheet.appendRow([novoId, novoCodigo, original.motor, now(), "Em processo", etapaInicial, etapaInicial, d.operador_nome || "", d.matricula || "", "", false, false, original.id_modelo || "", original.kit_qr || "", `Republicada limpa a partir de ${original.codigo_os}`, "", "", now()]);

  const subMap = {};
  let nextSub = nextId(subs.rows, "id_subtarefa");
  subs.rows.filter(s => String(s.id_os) === String(original.id_os) && !isTrue(s.excluida)).forEach(s => {
    const idNovoSub = nextSub++;
    subMap[String(s.id_subtarefa)] = idNovoSub;
    subs.sheet.appendRow([idNovoSub, novoId, novoCodigo, s.descricao, s.setor_origem, s.setor_destino, "Pendente", d.operador_nome || "", now(), "", "", "", false, "", ""]);
  });

  let nextItem = nextId(check.rows, "id_item");
  check.rows.filter(i => String(i.id_os) === String(original.id_os)).forEach(i => {
    const idSubNovo = i.id_subtarefa ? (subMap[String(i.id_subtarefa)] || "") : "";
    check.sheet.appendRow([nextItem++, novoId, i.origem, idSubNovo, i.etapa, i.setor, i.descricao, false, i.ordem, "", "", ""]);
  });

  addLog("OS_REPUBLICADA", d.operador_nome, `${original.codigo_os} -> ${novoCodigo} (limpa)`, novoCodigo);
  addHistorico("OS_REPUBLICADA", d.operador_nome, d.matricula, novoCodigo, etapaInicial, `Republicada limpa a partir de ${original.codigo_os}`);
  return jsonOut({sucesso:true, id_os:novoId, codigo_os:novoCodigo});
}

 * PRODUTIVIDADE TOTAL — V3.2
 ****************************************************/
function produtividadeResumo(matricula) {
  const osRows = rowsAsObjects(SHEETS.os).rows.filter(o => !isTrue(o.excluida));
  const checks = rowsAsObjects(SHEETS.checklist).rows;
  const subs = rowsAsObjects(SHEETS.subtarefas).rows.filter(s => !isTrue(s.excluida));
  const hist = rowsAsObjects(SHEETS.historico).rows;
  const logs = rowsAsObjects(SHEETS.logs).rows;

  const totalItens = checks.length;
  const concluidos = checks.filter(c => isTrue(c.concluido)).length;
  const abertas = osRows.filter(o => !isDone(o.status)).length;
  const subPendentes = subs.filter(s => !isDone(s.status)).length;

  const progresso = totalItens ? Math.round((concluidos / totalItens) * 100) : 0;
  const score = Math.max(0, Math.min(100, Math.round(progresso + 20 - Math.min(24, abertas * 3) - Math.min(18, subPendentes * 4))));

  const operadoresMap = {};
  hist.forEach(h => {
    const nome = norm(h.operador_nome);
    if (!nome || lower(nome).indexOf("admin") >= 0 || lower(nome).indexOf("gestor") >= 0) return;
    if (!operadoresMap[nome]) operadoresMap[nome] = {nome:nome, setor:h.setor || "", total:0, acao:"ações"};
    operadoresMap[nome].total++;
    if (h.setor) operadoresMap[nome].setor = h.setor;
  });

  const ranking = Object.keys(operadoresMap)
    .map(k => operadoresMap[k])
    .sort((a,b)=>b.total-a.total)
    .slice(0,8);

  const setorMap = {};
  subs.filter(s => !isDone(s.status)).forEach(s => {
    const setor = norm(s.setor_destino) || "Sem setor";
    if (!setorMap[setor]) setorMap[setor] = {setor:setor, total:0};
    setorMap[setor].total++;
  });

  const gargalos = Object.keys(setorMap)
    .map(k => setorMap[k])
    .sort((a,b)=>b.total-a.total)
    .slice(0,8);

  const alertas = [];
  if (subPendentes > 0) alertas.push(subPendentes + " subtarefa(s) pendente(s) podem bloquear avanço de OS.");
  if (abertas > 5) alertas.push("Fila elevada: " + abertas + " OS abertas no sistema.");
  if (score < 50) alertas.push("Produtividade crítica: revisar gargalos e pendências por setor.");

  const hoje = new Date();
  const logsHoje = logs.filter(l => {
    const d = new Date(l.data_hora);
    return !isNaN(d.getTime()) && d.getFullYear() === hoje.getFullYear() && d.getMonth() === hoje.getMonth() && d.getDate() === hoje.getDate();
  });

  return jsonOut({
    score: score,
    progresso: progresso,
    ranking_operadores: ranking,
    gargalos_setor: gargalos,
    alertas: alertas,
    eventos_hoje: logsHoje.length,
    resumo: {
      os_abertas: abertas,
      subtarefas_pendentes: subPendentes,
      itens_total: totalItens,
      itens_concluidos: concluidos
    }
  });
}

function temPermissao(matricula, chave) {
  const ops = rowsAsObjects(SHEETS.operadores).rows;
  const op = ops.find(o => norm(o.matricula) === norm(matricula));

  if (lower(op.perfil) === "admin") return true;

/****************************************************
 * KPI DASHBOARD AVANÇADO — V3.11
 ****************************************************/
function kpiDashboardAvancado(matricula) {
  const osRows = rowsAsObjects(SHEETS.os).rows.filter(o => !isTrue(o.excluida));
  const checks = rowsAsObjects(SHEETS.checklist).rows;
  const subs = rowsAsObjects(SHEETS.subtarefas).rows.filter(s => !isTrue(s.excluida));
  const hist = rowsAsObjects(SHEETS.historico).rows;
  const logs = rowsAsObjects(SHEETS.logs).rows;
  const operadores = rowsAsObjects(SHEETS.operadores).rows;

  const abertas = osRows.filter(o => !isDone(o.status)).length;
  const concluidas = osRows.filter(o => isDone(o.status)).length;
  const subPendentes = subs.filter(s => !isDone(s.status)).length;
  const totalItens = checks.length;
  const itensConcluidos = checks.filter(c => isTrue(c.concluido)).length;
  const taxaConclusao = totalItens ? Math.round((itensConcluidos / totalItens) * 100) : 0;

  const agora = new Date();
  const osAtrasadas = osRows.filter(o => {
    if (isDone(o.status)) return false;
    const d = new Date(o.data_abertura);
    if (isNaN(d.getTime())) return false;
    const horas = (agora - d) / 36e5;
    return horas >= 24;
  }).length;

  const hojeLogs = logs.filter(l => {
    const d = new Date(l.data_hora);
    return !isNaN(d.getTime()) && d.getFullYear() === agora.getFullYear() && d.getMonth() === agora.getMonth() && d.getDate() === agora.getDate();
  }).length;

  const setorMap = {};
  subs.filter(s => !isDone(s.status)).forEach(s => {
    const setor = norm(s.setor_destino) || "Sem setor";
    if (!setorMap[setor]) setorMap[setor] = {setor:setor, total:0};
    setorMap[setor].total++;
  });
  const setores = Object.keys(setorMap).map(k => setorMap[k]).sort((a,b)=>b.total-a.total);

  const operadorMap = {};
  hist.forEach(h => {
    const nome = norm(h.operador_nome);
    if (!nome || lower(nome).indexOf("admin") >= 0 || lower(nome).indexOf("gestor") >= 0) return;
    if (!operadorMap[nome]) operadorMap[nome] = {nome:nome, total:0, setor:h.setor || ""};
    operadorMap[nome].total++;
    if (h.setor) operadorMap[nome].setor = h.setor;
  });
  const rankingOps = Object.keys(operadorMap).map(k=>operadorMap[k]).sort((a,b)=>b.total-a.total);

  const score = Math.max(0, Math.min(100,
    Math.round(taxaConclusao + 20 - Math.min(30, osAtrasadas * 5) - Math.min(25, subPendentes * 3) - Math.min(15, abertas * 1.5))
  ));

  const alertas = [];
  if (osAtrasadas > 0) alertas.push(osAtrasadas + " OS em atraso exigem prioridade.");
  if (subPendentes > 0) alertas.push(subPendentes + " subtarefa(s) pendente(s) podem bloquear fluxo.");
  if (setores[0]) alertas.push("Maior gargalo atual: " + setores[0].setor + ".");
  if (score < 50) alertas.push("Saúde operacional crítica: revisar fila, atrasos e pendências.");

  return jsonOut({
    kpis: {
      score_operacional: score,
      os_abertas: abertas,
      os_concluidas: concluidas,
      os_atrasadas: osAtrasadas,
      subtarefas_pendentes: subPendentes,
      itens_total: totalItens,
      itens_concluidos: itensConcluidos,
      taxa_conclusao: taxaConclusao,
      eventos_hoje: hojeLogs,
      setor_critico: setores[0] ? setores[0].setor : "",
      gargalo_principal: setores[0] ? setores[0].setor : "",
      operador_destaque: rankingOps[0] ? rankingOps[0].nome : "",
      usuarios_ativos: operadores.filter(o => isTrue(o.ativo)).length
    },
    setores: setores,
    operadores: rankingOps,
    alertas: alertas,
    atualizado_em: now()
  });
}

function auditoriaEstruturaInternaV4() {
  const schemas = schemasBancoV4();
  return Object.keys(schemas).map(name => {
    const sheet = sh(name);
    const lastRow = sheet.getLastRow();
    const lastCol = sheet.getLastColumn();
    const expected = schemas[name];
    const headers = lastRow > 0 && lastCol > 0
      ? sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => norm(h))
      : [];
    const matches = headers.filter(h => expected.indexOf(h) >= 0).length;
    const missing = expected.filter(h => headers.indexOf(h) < 0);
    const primeiraLinhaVazia = headers.every(h => !h);
    const semCabecalho = lastRow === 0 || matches === 0;
    return {
      aba:name,
      linhas:lastRow,
      colunas:lastCol,
      ok:missing.length === 0 && !semCabecalho,
      sem_cabecalho:semCabecalho,
      primeira_linha_vazia:primeiraLinhaVazia,
      campos_presentes:matches,
      campos_ausentes:missing,
      acao_sugerida:semCabecalho ? "inserir_cabecalho" : (missing.length ? "adicionar_colunas" : "nenhuma")
    };
  });
}

function corrigirCabecalhosV4(d) {
  const op = exigirPerfil(d.matricula, ["Admin"]);
  if (norm(d.confirmacao) !== "CORRIGIR_CABECALHOS") {
    return jsonOut({sucesso:false, erro:"Confirmacao obrigatoria: CORRIGIR_CABECALHOS"});
  }

  const schemas = schemasBancoV4();
  const auditoria = auditoriaEstruturaInternaV4();
  const alteracoes = [];

  auditoria.forEach(item => {
    if (item.ok) return;
    const sheet = sh(item.aba);
    const expected = schemas[item.aba];

    if (item.sem_cabecalho) {
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(expected);
      } else if (item.primeira_linha_vazia) {
        sheet.getRange(1, 1, 1, expected.length).setValues([expected]);
      } else {
        sheet.insertRowBefore(1);
        sheet.getRange(1, 1, 1, expected.length).setValues([expected]);
      }
      sheet.setFrozenRows(1);
      alteracoes.push({aba:item.aba, acao:"cabecalho_inserido", campos:expected.length});
      return;
    }

    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(h => norm(h));
    let colIndex = headers.length + 1;
    item.campos_ausentes.forEach(field => {
      sheet.getRange(1, colIndex).setValue(field);
      colIndex++;
    });
    sheet.setFrozenRows(1);
    alteracoes.push({aba:item.aba, acao:"colunas_adicionadas", campos:item.campos_ausentes});
  });

  addLog("CABECALHOS_V4_CORRIGIDOS", op.nome, alteracoes.length + " aba(s) ajustadas", "");
  addHistorico("CABECALHOS_V4_CORRIGIDOS", op.nome, op.matricula, "", "Administracao", alteracoes.length + " aba(s) ajustadas");
  return jsonOut({sucesso:true, alteracoes:alteracoes, atualizado_em:now()});
}



/****************************************************
 * KPI INTELIGENTE V4
 * Analise operacional preditiva baseada nas abas atuais.
 ****************************************************/
function parseDateV4(v) {
  if (!v) return null;
  if (Object.prototype.toString.call(v) === "[object Date]") {
    return isNaN(v.getTime()) ? null : v;
  }

  if (typeof v === "number" && isFinite(v)) {
    if (v > 20000 && v < 80000) {
      return new Date(Math.round((v - 25569) * 86400000));
    }
    return null;
  }

  const raw = String(v || "").trim();
  if (!raw) return null;
  if (/^\d{1,7}:\d{2}:\d{2}$/.test(raw)) return null;

  if (/^\d+(\.\d+)?$/.test(raw)) {
    const serial = Number(raw);
    if (serial > 20000 && serial < 80000) {
      return new Date(Math.round((serial - 25569) * 86400000));
    }
  }

  const br = raw.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})(?:[,\sT]+(\d{1,2}):(\d{2})(?::(\d{2}))?)?$/);
  if (br) {
    const d = new Date(Number(br[3]), Number(br[2]) - 1, Number(br[1]), Number(br[4] || 0), Number(br[5] || 0), Number(br[6] || 0));
    return isNaN(d.getTime()) ? null : d;
  }

  const d = new Date(raw);
  return isNaN(d.getTime()) ? null : d;
}

function isHojeV4(v) {
  const d = parseDateV4(v);
  if (!d) return false;
  const h = new Date();
  return d.getFullYear() === h.getFullYear() && d.getMonth() === h.getMonth() && d.getDate() === h.getDate();
}

function hoursBetweenV4(a, b) {
  const start = parseDateV4(a);
  const end = parseDateV4(b);
  if (!start || !end) return null;
  return Math.max(0, (end - start) / 36e5);
}

function ageHoursV4(v) {
  const d = parseDateV4(v);
  if (!d) return 0;
  return Math.max(0, (new Date() - d) / 36e5);
}

function isWithinDaysV4(v, days) {
  const d = parseDateV4(v);
  if (!d) return false;
  return (new Date() - d) <= (Number(days || 0) * 24 * 36e5);
}

function avgV4(values) {
  const nums = values.filter(v => typeof v === "number" && !isNaN(v));
  if (!nums.length) return 0;
  return nums.reduce((a,b)=>a+b,0) / nums.length;
}

function pctV4(done, total) {
  total = Number(total || 0);
  if (!total) return 0;
  return Math.round((Number(done || 0) / total) * 100);
}

function round1V4(v) {
  return Math.round(Number(v || 0) * 10) / 10;
}

function numberV4(v, fallback) {
  const n = Number(v);
  return isFinite(n) ? n : Number(fallback || 0);
}

function contarDatasInvalidasV4(rows, fields) {
  let total = 0;
  (rows || []).forEach(r => {
    (fields || []).forEach(f => {

    });
  });
  return total;
}

function qualidadeDadosKpiV4(dados) {
  const esperado = {};
  esperado[SHEETS.os] = ["id_os","codigo_os","motor","data_abertura","status","etapa_atual","setor_atual","concluida_em","ultimo_movimento_em"];
  esperado[SHEETS.checklist] = ["id_item","id_os","descricao","concluido","concluido_em","tempo_item_min"];
  esperado[SHEETS.subtarefas] = ["id_subtarefa","id_os","codigo_os","descricao","setor_destino","status","criado_em","concluido_em","tempo_subtarefa_min"];
  esperado[SHEETS.historico] = ["data_hora","acao","operador_nome","matricula","codigo_os","setor","detalhes"];
  esperado[SHEETS.logs] = ["data_hora","acao","operador_nome","codigo_os","detalhes"];
  esperado[SHEETS.setores] = ["id_setor","nome_setor","codigo_setor","tipo","ativo"];
  esperado[SHEETS.fluxos] = ["id_fluxo","nome_fluxo","etapa","ordem","ativo"];
  esperado[SHEETS.config_sistema] = ["chave","valor","atualizado_por","atualizado_em"];

  const abas = Object.keys(esperado).map(nome => {
    const faltantes = headersFaltantesV4(nome, esperado[nome]);
    return {aba:nome, campos_ausentes:faltantes, ok:faltantes.length === 0};
  }).filter(item => !item.ok);

  const datasInvalidas =
    contarDatasInvalidasV4(dados.osRows, ["data_abertura","concluida_em","ultimo_movimento_em"]) +
    contarDatasInvalidasV4(dados.checks, ["concluido_em"]) +
    contarDatasInvalidasV4(dados.subs, ["criado_em","concluido_em","excluida_em"]) +
    contarDatasInvalidasV4(dados.hist, ["data_hora"]) +
    contarDatasInvalidasV4(dados.logs, ["data_hora"]);

  return {
    score_qualidade: Math.max(0, 100 - Math.min(45, abas.length * 9) - Math.min(35, datasInvalidas * 4)),
    datas_invalidas: datasInvalidas,
    abas_inconsistentes: abas,
    observacao: abas.length || datasInvalidas ? "Ha dados que devem ser revisados na planilha." : "Estrutura principal validada."
  };
}

function kpiInteligenteV4(matricula) {
  const osRows = rowsAsObjects(SHEETS.os).rows.filter(o => !isTrue(o.excluida));
  const checks = rowsAsObjects(SHEETS.checklist).rows;
  const subs = rowsAsObjects(SHEETS.subtarefas).rows.filter(s => !isTrue(s.excluida));
  const hist = rowsAsObjects(SHEETS.historico).rows;
  const logs = rowsAsObjects(SHEETS.logs).rows;
  const slaHoras = Math.max(1, numberV4(configValorV4("tempo_alerta_horas", 24), 24));
  const faixaAtencao = Math.max(2, slaHoras / 3);
  const faixaAlta = Math.max(faixaAtencao + 1, slaHoras * 2 / 3);
  const qualidade = qualidadeDadosKpiV4({osRows:osRows, checks:checks, subs:subs, hist:hist, logs:logs});

  const abertas = osRows.filter(o => !isDone(o.status));
  const concluidas = osRows.filter(o => isDone(o.status));
  const subsPendentes = subs.filter(s => !isDone(s.status));
  const checksConcluidos = checks.filter(c => isTrue(c.concluido));
  const checksHoje = checksConcluidos.filter(c => isHojeV4(c.concluido_em));
  const checks7d = checksConcluidos.filter(c => isWithinDaysV4(c.concluido_em, 7));

  const leadTimes = concluidas
    .map(o => hoursBetweenV4(o.data_abertura, o.concluida_em))
    .filter(v => typeof v === "number" && !isNaN(v));

  const subtarefaTimes = subs
    .filter(s => isDone(s.status))
    .map(s => {
      const minutos = numberV4(s.tempo_subtarefa_min, 0);
      if (minutos > 0) return minutos / 60;
      return hoursBetweenV4(s.criado_em, s.concluido_em);
    })
    .filter(v => typeof v === "number" && !isNaN(v));

  const abertasComRisco = abertas.map(o => {
    const progresso = calcularProgresso(o.id_os);
    const idade = ageHoursV4(o.ultimo_movimento_em || o.data_abertura);
    const idadeTotal = ageHoursV4(o.data_abertura);
    let risco = 0;
    if (idadeTotal >= slaHoras) risco += 42;
    else if (idadeTotal >= faixaAlta) risco += 28;
    else if (idadeTotal >= faixaAtencao) risco += 14;
    if (idade >= faixaAtencao) risco += 18;
    if (progresso.percentual_total < 25 && idadeTotal >= faixaAtencao) risco += 18;
    if (Number(progresso.subtarefas_pendentes || 0) > 0) risco += Math.min(20, Number(progresso.subtarefas_pendentes || 0) * 8);
    return {
      id_os:o.id_os,
      codigo_os:o.codigo_os,
      motor:o.motor,
      setor:o.setor_atual,
      etapa:o.etapa_atual,
      status:o.status,
      progresso:progresso.percentual_total,
      idade_horas:round1V4(idadeTotal),
      parado_horas:round1V4(idade),
      subtarefas_pendentes:progresso.subtarefas_pendentes,
      risco:Math.max(0, Math.min(100, Math.round(risco)))
    };
  }).sort((a,b)=>b.risco-a.risco || b.idade_horas-a.idade_horas);

  const setorMap = {};
  abertasComRisco.forEach(o => {
    const item = pushMapV4(setorMap, o.setor);
    item.os_abertas++;
    item._idades.push(o.idade_horas);
    item.peso += 2 + Math.min(8, o.idade_horas / 8) + Math.min(8, o.risco / 18);
  });
  subsPendentes.forEach(s => {
    const item = pushMapV4(setorMap, s.setor_destino);
    item.subtarefas_pendentes++;
    item._idades.push(ageHoursV4(s.criado_em));
    item.peso += 3 + Math.min(10, ageHoursV4(s.criado_em) / 8);
  });

  const setores = Object.keys(setorMap).map(k => {
    const item = setorMap[k];
    item.idade_media_horas = round1V4(avgV4(item._idades));
    item.peso = Math.round(item.peso);
    delete item._idades;
    return item;
  }).sort((a,b)=>b.peso-a.peso);

  const acoesCriticas = hist.concat(logs).filter(r => {
    const acao = lower(r.acao);
    return acao.indexOf("subtarefa") >= 0 || acao.indexOf("republicar") >= 0 || acao.indexOf("reiniciar") >= 0 || acao.indexOf("exclu") >= 0;
  });
  const retrabalho7d = acoesCriticas.filter(r => isWithinDaysV4(r.data_hora, 7)).length;

  const operadorHoje = {};
  hist.filter(h => isHojeV4(h.data_hora)).forEach(h => {
    const key = norm(h.matricula) || norm(h.operador_nome);
    if (key) operadorHoje[key] = true;
  });

  const gargalo = setores[0] || {setor:"", peso:0, os_abertas:0, subtarefas_pendentes:0};
  const osRisco = abertasComRisco.filter(o => o.risco >= 45).slice(0,8);
  const taxaSla = pctV4(leadTimes.filter(h => h <= slaHoras).length, leadTimes.length);
  const idadeMediaFila = round1V4(avgV4(abertasComRisco.map(o => o.idade_horas)));
  const scoreBase = JSON.parse(kpiDashboardAvancado(matricula).getContent()).kpis || {};
  const score = Math.max(0, Math.min(100,
    Math.round(Number(scoreBase.score_operacional || 0) - Math.min(18, osRisco.length * 4) - Math.min(12, retrabalho7d * 1.5) - Math.min(12, (100 - qualidade.score_qualidade) / 8) + Math.min(8, taxaSla / 16))
  ));

  const alertas = [];
  if (osRisco.length) alertas.push(osRisco.length + " OS em risco operacional. Priorize " + osRisco[0].codigo_os + ".");
  if (gargalo.setor) alertas.push("Gargalo principal: " + gargalo.setor + " com peso " + gargalo.peso + ".");
  if (idadeMediaFila >= faixaAlta) alertas.push("Fila envelhecida: media de " + idadeMediaFila + "h nas OS abertas.");
  if (retrabalho7d >= 3) alertas.push("Retrabalho alto nos ultimos 7 dias: " + retrabalho7d + " eventos criticos.");
  if (qualidade.datas_invalidas) alertas.push(qualidade.datas_invalidas + " campo(s) de data precisam revisao na planilha.");
  if (qualidade.abas_inconsistentes.length) alertas.push(qualidade.abas_inconsistentes.length + " aba(s) com cabecalho incompleto.");
  if (!alertas.length) alertas.push("Fluxo estavel: nenhum risco critico detectado agora.");

  const recomendacoes = [];
  if (osRisco[0]) recomendacoes.push("Abrir " + osRisco[0].codigo_os + " e remover bloqueios antes de novas OS.");
  if (gargalo.setor) recomendacoes.push("Rebalancear equipe ou prioridade para " + gargalo.setor + ".");
  if (subsPendentes.length) recomendacoes.push("Fechar subtarefas pendentes antes de avancar etapas dependentes.");
  if (checksHoje.length === 0 && abertas.length) recomendacoes.push("Checar aderencia do checklist: nenhuma conclusao registrada hoje.");
  if (qualidade.abas_inconsistentes.length) recomendacoes.push("Corrigir cabecalhos de " + qualidade.abas_inconsistentes.map(a => a.aba).slice(0,3).join(", ") + " antes de liberar novos indicadores.");

  return jsonOut({
    kpis:{
      score_inteligente:score,
      score_qualidade_dados:qualidade.score_qualidade,
      lead_time_medio_horas:round1V4(avgV4(leadTimes)),
      tempo_medio_subtarefa_horas:round1V4(avgV4(subtarefaTimes)),
      sla_24h:taxaSla,
      sla_configurado:taxaSla,
      sla_configurado_horas:slaHoras,
      idade_media_fila_horas:idadeMediaFila,
      os_em_risco:osRisco.length,
      retrabalho_7d:retrabalho7d,
      throughput_hoje:checksHoje.length,
      throughput_7d:checks7d.length,
      operadores_ativos_hoje:Object.keys(operadorHoje).length,
      gargalo_principal:gargalo.setor || ""
    },
    os_risco:osRisco,
    setores:setores,
    qualidade_dados:qualidade,
    alertas:alertas,
    recomendacoes:recomendacoes,
    atualizado_em:now()
  });
}

/****************************************************
 * PERFORMANCE API — V3.15
 * Reduz leituras repetidas e prepara crescimento.
 ****************************************************/
function cacheJson_(key, seconds, fn) {
  const cache = CacheService.getScriptCache();
  const hit = cache.get(key);
  if (hit) return jsonOut(JSON.parse(hit));

  const data = fn();
  try {
    cache.put(key, JSON.stringify(data), seconds || 20);
  } catch(e) {}
  return jsonOut(data);
}

function dashboardLean(matricula) {
  return cacheJson_("dashboardLean:" + norm(matricula), 15, function(){
    const kpi = JSON.parse(kpiDashboardAvancado(matricula).getContent());
    const logs = rowsAsObjects(SHEETS.logs).rows
      .sort((a,b)=>new Date(b.data_hora)-new Date(a.data_hora))
      .slice(0,30);
    const notif = rowsAsObjects(SHEETS.notificacoes).rows
      .filter(n => norm(n.destino) === "Admin" && !isTrue(n.lida))
      .length;

    return {
      kpis:kpi.kpis || {},
      setores:kpi.setores || [],
      operadores:kpi.operadores || [],
      alertas:kpi.alertas || [],
      logs:logs,
      notificacoes_nao_lidas:notif,
      atualizado_em:now()
    };
  });
}

function bootstrapApp(matricula) {
  return cacheJson_("bootstrapApp:" + norm(matricula), 12, function(){
    const op = rowsAsObjects(SHEETS.operadores).rows.find(o => norm(o.matricula) === norm(matricula));
    if (!op) return {operador:null, erro:"Operador não encontrado"};

    const perfil = lower(op.perfil);
    const base = {
      operador:op,
      perfil:op.perfil,
      setor:op.setor,
      atualizado_em:now()
    };

    if (perfil === "admin") {
      const dash = JSON.parse(dashboardLean(matricula).getContent());
      return Object.assign(base, {dashboard:dash});
    }

    if (perfil === "gestao" || perfil === "gestor") {
      const resumo = JSON.parse(gestorResumo(matricula).getContent());
      const kpi = JSON.parse(kpiDashboardAvancado(matricula).getContent());
      return Object.assign(base, {resumo:resumo, dashboard:kpi});
    }

    const resumo = JSON.parse(gestorResumo(matricula).getContent());
    return Object.assign(base, {resumo:resumo});
  });
}

function gestorAplicarModeloChecklist(d) {
  const osData = rowsAsObjects(SHEETS.os);
  const os = osData.rows.find(o => String(o.id_os) === String(d.id_os));
  if (!os) return jsonOut({sucesso:false, erro:"OS não encontrada"});

  const modelos = rowsAsObjects(SHEETS.modelos).rows;
  const modelo = modelos.find(m => String(m.id_modelo) === String(d.id_modelo));
  if (!modelo) return jsonOut({sucesso:false, erro:"Modelo não encontrado"});

  const setor = modelo.setor || os.setor_atual || os.etapa_atual || "Operação";
  const check = rowsAsObjects(SHEETS.checklist);
  const atuais = check.rows
    .filter(i => String(i.id_os) === String(os.id_os))
    .filter(i => norm(i.origem) === "Principal")
    .filter(i => norm(i.setor || i.etapa) === norm(setor));

  if (String(d.modo || "adicionar").toLowerCase() === "substituir") {
    atuais
      .filter(i => !isTrue(i.concluido))
      .forEach(i => setCell(check.sheet, check.headers, i.__rowNumber, "concluido", true));
  }

  const modelosDoSetor = modelos
    .filter(m => isTrue(m.ativo))
    .filter(m => lower(m.tipo) === "principal")
    .filter(m => norm(m.setor) === norm(setor))
    .sort((a,b)=>Number(a.ordem||0)-Number(b.ordem||0));

  const base = modelosDoSetor.length ? modelosDoSetor : [modelo];
  let idItem = nextId(check.rows, "id_item");
  let ordem = atuais.length + 1;
  let criados = 0;

  base.forEach(m => {
    const descricao = m.descricao || modelo.descricao || "Item do modelo";
    const existe = check.rows.some(i =>
      String(i.id_os) === String(os.id_os) &&
      norm(i.origem) === "Principal" &&
      norm(i.setor || i.etapa) === norm(setor) &&
      norm(i.descricao) === norm(descricao) &&
      !isTrue(i.concluido)
    );

    if (existe && String(d.modo || "adicionar").toLowerCase() !== "substituir") return;

    check.sheet.appendRow([idItem++, os.id_os, "Principal", "", setor, setor, descricao, false, ordem++, "", "", ""]);
    criados++;
  });

  ensureColumn(SHEETS.os, "id_modelo");
  setCellSafe(SHEETS.os, os.__rowNumber, "id_modelo", d.id_modelo);

  addHistorico("CHECKLIST_MODELO_APLICADO", d.operador_nome || "Gestão", d.matricula || "", os.codigo_os, setor, `Modelo ${d.id_modelo} • ${criados} item(ns)`);
  addLog("CHECKLIST_MODELO_APLICADO", d.operador_nome || "Gestão", `${criados} item(ns)`, os.codigo_os);

  return jsonOut({sucesso:true, criados:criados});
}

  row.id_os = d.id_os || "";
  row.descricao = d.descricao || "";
  row.setor_destino = d.setor_destino || "";
  row.status = "Pendente";
  row.excluida = false;
  row.criado_por = d.operador_nome || "Gestão";
  row.criado_em = now();

  data.sheet.appendRow(headers.map(h => row[h] !== undefined ? row[h] : ""));

  addHistorico("SUBTAREFA_CRIADA_GESTAO", d.operador_nome || "Gestão", d.matricula || "", d.id_os || "", d.setor_destino || "", d.descricao || id);
  addLog("SUBTAREFA_CRIADA_GESTAO", d.operador_nome || "Gestão", d.descricao || id, d.id_os || "");

  return jsonOut({sucesso:true, id_subtarefa:id});
}

function gestorEditarSubtarefa(d) {
  const data = rowsAsObjects(SHEETS.subtarefas);
  const row = data.rows.find(s => String(s.id_subtarefa) === String(d.id_subtarefa));

  ensureColumn(SHEETS.subtarefas, "alterado_por");
  ensureColumn(SHEETS.subtarefas, "alterado_em");

  if (d.descricao !== undefined) setCellSafe(SHEETS.subtarefas, row.__rowNumber, "descricao", d.descricao);
  if (d.setor_destino !== undefined) setCellSafe(SHEETS.subtarefas, row.__rowNumber, "setor_destino", d.setor_destino);
  if (d.status !== undefined) setCellSafe(SHEETS.subtarefas, row.__rowNumber, "status", d.status);

  setCellSafe(SHEETS.subtarefas, row.__rowNumber, "alterado_por", d.operador_nome || "Gestão");
  setCellSafe(SHEETS.subtarefas, row.__rowNumber, "alterado_em", now());

  addHistorico("SUBTAREFA_EDITADA_GESTAO", d.operador_nome || "Gestão", d.matricula || "", row.id_os || "", d.setor_destino || row.setor_destino || "", d.descricao || row.descricao || row.id_subtarefa);
  addLog("SUBTAREFA_EDITADA_GESTAO", d.operador_nome || "Gestão", row.id_subtarefa, row.id_os || "");

  return jsonOut({sucesso:true});
}

function gestorCriarSequenciaSubtarefas(d) {
  const setores = String(d.setores || "")
    .split(",")
    .map(s => s.trim())
    .filter(Boolean);

  if (!setores.length) return jsonOut({sucesso:false, erro:"Nenhum setor informado"});

  const criadas = [];
  setores.forEach((setor, index) => {
    const desc = `${index + 1}. ${d.descricao || "Subtarefa sequencial"} — ${setor}`;
    const res = gestorCriarSubtarefa({
      id_os:d.id_os || "",
      setor_destino:setor,
      descricao:desc,
      operador_nome:d.operador_nome || "Gestão",
      matricula:d.matricula || ""
    });
    criadas.push({setor:setor, descricao:desc});
  });

  addHistorico("SEQUENCIA_SUBTAREFAS_CRIADA", d.operador_nome || "Gestão", d.matricula || "", d.id_os || "", setores.join(" > "), d.descricao || "");
  addLog("SEQUENCIA_SUBTAREFAS_CRIADA", d.operador_nome || "Gestão", setores.join(" > "), d.id_os || "");

  return jsonOut({sucesso:true, criadas:criadas});
}

function gestorAplicarModeloSubtarefa(d) {
  const modelos = {
    eletrica:["Verificar continuidade elétrica","Medir isolação","Registrar condição dos cabos"],
    usinagem:["Avaliar desgaste dimensional","Definir correção de usinagem","Registrar medição final"],
    qualidade:["Inspecionar conformidade","Validar rastreabilidade","Liberar ou apontar pendência"],
    montagem:["Separar componentes","Apoiar montagem","Validar fechamento da etapa"]
  };

  const lista = modelos[String(d.modelo || "").toLowerCase()] || modelos.eletrica;
  const criadas = [];

  lista.forEach(desc => {
    gestorCriarSubtarefa({
      id_os:d.id_os || "",
      setor_destino:d.setor_destino || "",
      descricao:desc,
      operador_nome:d.operador_nome || "Gestão",
      matricula:d.matricula || ""
    });
    criadas.push(desc);
  });

  if (d.id_subtarefa) {
    gestorEditarSubtarefa({
      id_subtarefa:d.id_subtarefa,
      status:"Em processo",
      operador_nome:d.operador_nome || "Gestão",
      matricula:d.matricula || ""
    });
  }

  addHistorico("MODELO_SUBTAREFA_APLICADO", d.operador_nome || "Gestão", d.matricula || "", d.id_os || "", d.setor_destino || "", d.modelo || "");
  addLog("MODELO_SUBTAREFA_APLICADO", d.operador_nome || "Gestão", d.modelo || "", d.id_os || "");

  return jsonOut({sucesso:true, criadas:criadas});
}

function gestorRepetirSubtarefa(d) {
  const data = rowsAsObjects(SHEETS.subtarefas);
  const original = data.rows.find(s => String(s.id_subtarefa) === String(d.id_subtarefa));

  const idOs = d.id_os || (original ? original.id_os : "");
  const setor = d.setor_destino || (original ? original.setor_destino : "");

  if (!descricao) return jsonOut({sucesso:false, erro:"Descrição não informada"});

    setor_destino:setor,
    descricao:descricao,
    operador_nome:d.operador_nome || "Gestão",

  });

  addHistorico("SUBTAREFA_REPETIDA_GESTAO", d.operador_nome || "Gestão", d.matricula || "", idOs || "", setor || "", descricao);
  addLog("SUBTAREFA_REPETIDA_GESTAO", d.operador_nome || "Gestão", descricao, idOs || "");

  return jsonOut({sucesso:true});
}



/****************************************************
 * ADMIN — PERMISSÕES DA GESTÃO V3.34
 * Preparado para persistência em planilha "permissoes_gestao".
 ****************************************************/
function adminSalvarPermissoesGestao(d) {
  ensureSheet("permissoes_gestao", ["chave","liberado","alterado_por","alterado_em"]);
  const sheet = sh("permissoes_gestao");
  const payload = typeof d.permissoes === "string" ? JSON.parse(d.permissoes || "{}") : (d.permissoes || {});
  sheet.clearContents();
  sheet.appendRow(["chave","liberado","alterado_por","alterado_em"]);
  Object.keys(payload).forEach(k => {
    sheet.appendRow([k, payload[k] !== false, d.operador_nome || "Admin", now()]);
  });
  addLog("PERMISSOES_GESTAO_ATUALIZADAS", d.operador_nome || "Admin", "Permissões da Gestão atualizadas", "");
  return jsonOut({sucesso:true});
}

function adminPermissoesGestao() {
  ensureSheet("permissoes_gestao", ["chave","liberado","alterado_por","alterado_em"]);
  const rows = rowsAsObjects("permissoes_gestao").rows;
  const permissoes = {};
  rows.forEach(r => {
    if(r.chave) permissoes[r.chave] = isTrue(r.liberado);
  });
  return jsonOut({permissoes:permissoes});
}
